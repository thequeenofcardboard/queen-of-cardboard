function checkOAuthReturn(){
  const params = new URLSearchParams(window.location.search);
  const status = params.get('verify_status');
  if(!status) return;
  // Clean the URL
  window.history.replaceState({}, document.title, window.location.pathname);

  const platform = params.get('platform') || '';
  const name     = params.get('name') || '';
  const profile  = params.get('profile') || '';
  const ig       = params.get('ig') || '';

  const box = document.getElementById('connect-status-box');

  if(status === 'pending'){
    box.style.display = 'block';
    const igLine = ig ? `<br>Instagram: <strong>@${ig}</strong>` : '';
    box.innerHTML = `<div class="verify-status-box pending">
      ✅ <strong>Verification request submitted!</strong><br>
      Platform: <strong>${platformLabel(platform)}</strong> — ${name}${igLine}<br>
      The Queen of Cardboard will review your account and email you an access code within 1–2 business days.
    </div>`;
    openModal();
    switchModalTab('connect', document.querySelector('.modal-tab'));
    // Mark the connected platform
    markPlatformConnected(platform);
  } else if(status === 'denied'){
    box.style.display = 'block';
    box.innerHTML = `<div class="verify-status-box error">❌ You declined to connect ${platformLabel(platform)}. You can try again or connect a different platform.</div>`;
    openModal();
  } else if(status === 'error'){
    box.style.display = 'block';
    box.innerHTML = `<div class="verify-status-box error">⚠ Something went wrong during verification. Please try again or contact us at <a href="mailto:thequeenofcardboard1@gmail.com">thequeenofcardboard1@gmail.com</a>.</div>`;
    openModal();
  }
}

async function submitWhatnotRequest(){
  const name  = document.getElementById('buyer-name-connect').value.trim();
  const email = document.getElementById('buyer-email-connect').value.trim();
  const url   = document.getElementById('whatnot-url').value.trim();
  const notes = document.getElementById('whatnot-notes').value.trim();

  if(!name || !email){ showConnectError('Please enter your name and email first.'); return; }
  if(!url){ showConnectError('Please enter your Whatnot profile URL.'); return; }
  if(!url.includes('whatnot.com')){ showConnectError('Please enter a valid whatnot.com URL.'); return; }

  const btn = document.querySelector('#whatnot-form .btn');
  btn.textContent = 'Submitting...';
  btn.disabled = true;

  try {
    const res = await fetch('/.netlify/functions/verify-whatnot', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ name, email, whatnotUrl: url, extraNotes: notes })
    });
    const data = await res.json();

    if(data.success){
      setStatus('whatnot-status','Pending ⏳','status-pending');
      document.getElementById('whatnot-form').classList.remove('open');
      const box = document.getElementById('connect-status-box');
      box.style.display = 'block';
      box.innerHTML = `<div class="verify-status-box pending">✅ <strong>Whatnot request submitted!</strong><br>${data.message}</div>`;
      showToast('Whatnot request submitted ✓');
    } else {
      showConnectError(data.error || 'Submission failed. Please try again.');
    }
  } catch(e){
    showConnectError('Network error. Please try again.');
  }

  btn.textContent = 'Submit Whatnot Request';
  btn.disabled = false;
}

function saveProduct(){
  const name=document.getElementById('ep-name').value.trim();
  const err=document.getElementById('ep-error');
  if(!name){ err.style.display='block'; err.textContent='❌ Product name is required.'; return; }
  const priceVal=document.getElementById('ep-price').value.trim();
  const product={
    id: editingProductId||('prod-'+Date.now()),
    category: document.getElementById('ep-cat').value,
    name, desc: document.getElementById('ep-desc').value.trim(),
    price: priceVal!==''?parseFloat(priceVal):null,
    image: document.getElementById('ep-image-data').value||'',
    badge: document.getElementById('ep-badge').value,
    inStock: document.getElementById('ep-instock').checked
  };
  if(editingProductId){ const i=PRODUCTS.findIndex(p=>p.id===editingProductId); if(i!==-1) PRODUCTS[i]=product; }
  else PRODUCTS.push(product);
  saveProducts(); renderProducts(); closeEditModal();
  showToast(editingProductId?'✓ Product updated!':'✓ Product added!');
}
function deleteProduct(){
  if(!editingProductId||!confirm('Delete this product? This cannot be undone.')) return;
  PRODUCTS=PRODUCTS.filter(p=>p.id!==editingProductId);
  saveProducts(); renderProducts(); closeEditModal();
  showToast('Product deleted.');
}

/* ── IMAGE HANDLING ─────────────────────────────── */
function handleImageUpload(event){
  const file=event.target.files[0]; if(!file) return;
  const reader=new FileReader();
  reader.onload=e=>{
    document.getElementById('ep-image-data').value=e.target.result;
    document.getElementById('ep-img-preview').innerHTML=`<img src="${e.target.result}" alt="">`;
  };
  reader.readAsDataURL(file);
}

/* ── PRODUCT RENDERING ──────────────────────────── */
function renderProducts(){
  const grid=document.getElementById('product-grid');
  const list=currentFilter==='all'?PRODUCTS:PRODUCTS.filter(p=>p.category===currentFilter);
  if(!list.length){ grid.innerHTML=`<p style="color:var(--text-muted);font-size:.9rem;text-align:center;grid-column:1/-1;padding:3rem 0">No products in this category yet — check back soon!</p>`; return; }
  grid.innerHTML=list.map(p=>{
    const priceDisplay=(verified||isOwner)?(p.price!==null?`$${Number(p.price).toFixed(2)}`:'Contact for Pricing'):'<span style="filter:blur(5px);user-select:none">$••••</span>';
    const editBtn=isOwner?`<button class="owner-edit-btn" onclick="openEditProduct('${p.id}')">✏ Edit Product</button>`:'';
    let footerBtn='';
    if(!p.inStock) footerBtn=`<span style="font-size:.72rem;color:var(--text-muted);font-family:var(--font-d);letter-spacing:.1em">Coming Soon</span>`;
    else if(verified||isOwner) footerBtn=`<button class="add-to-cart" onclick="addToCart('${p.id}')">Add to Order</button>`;
    else footerBtn=`<button class="add-to-cart locked" onclick="openModal()">🔒 Get Verified</button>`;
    return `<div class="product-card">
      ${p.badge&&p.inStock?`<span class="badge-new">${p.badge}</span>`:''}
      ${!p.inStock?`<span class="badge-sold">Coming Soon</span>`:''}
      ${p.image?`<img src="${p.image}" alt="${p.name}" class="product-img" loading="lazy" onerror="this.style.display='none'">`:`<div class="product-img-placeholder"><span>♛</span>Photo Coming Soon</div>`}
      <div class="product-body">
        <div class="product-category">${CATEGORY_LABELS[p.category]||p.category}</div>
        <div class="product-name">${p.name}</div>
        ${p.desc?`<div class="product-desc">${p.desc}</div>`:''}
        <div style="font-family:var(--font-d);font-size:.55rem;letter-spacing:.15em;color:var(--gold-dark);margin-bottom:.75rem;text-transform:uppercase">✅ Authentic &amp; Sealed</div>
        <div class="product-footer">
          <div class="product-price"><small>Price</small>${priceDisplay}</div>
          ${footerBtn}
        </div>
      </div>
      ${editBtn}
    </div>`;
  }).join('');
}

/* ── CART ───────────────────────────────────────── */
function addToCart(id){
  if(!verified&&!isOwner){ openModal(); return; }
  const p=PRODUCTS.find(x=>x.id===id); if(!p) return;
  const ex=cart.find(i=>i.id===id);
  if(ex) ex.qty++; else cart.push({id,name:p.name,price:p.price,qty:1});
  updateCartUI(); showToast('Added to order ✓');
}
function updateCartCount(){ document.getElementById('cart-count').textContent=cart.reduce((s,i)=>s+i.qty,0); }
function updateCartUI(){
  updateCartCount();
  const el=document.getElementById('cart-items');
  if(!cart.length){ el.innerHTML='<div class="cart-empty"><span>🛒</span>Your cart is empty</div>'; document.getElementById('cart-total').textContent='$0.00'; document.getElementById('shipping-note').textContent='Add $500+ to qualify for free shipping'; return; }
  let total=0,allPriced=true;
  el.innerHTML=cart.map(item=>{
    const lt=item.price?item.price*item.qty:null;
    if(lt) total+=lt; else allPriced=false;
    return `<div class="cart-item">
      <div class="cart-item-info">
        <div class="cart-item-name">${item.name}</div>
        <div class="cart-item-price">${item.price?'$'+(item.price*item.qty).toFixed(2):'Price TBD'}</div>
        <div class="cart-item-qty">
          <button class="qty-btn" onclick="changeQty('${item.id}',-1)">−</button>
          <span class="qty-display">${item.qty}</span>
          <button class="qty-btn" onclick="changeQty('${item.id}',1)">+</button>
        </div>
      </div>
      <button class="remove-item" onclick="removeFromCart('${item.id}')">✕</button>
    </div>`;
  }).join('');
  document.getElementById('cart-total').textContent=allPriced?`$${total.toFixed(2)}`:`$${total.toFixed(2)} + TBD`;
  const note=document.getElementById('shipping-note');
  note.textContent=total>=500?'✓ Free shipping applied!':'Add $'+(500-total).toFixed(2)+' more for free shipping';
}
function changeQty(id,d){ const i=cart.find(x=>x.id===id); if(!i) return; i.qty=Math.max(1,i.qty+d); updateCartUI(); }
function removeFromCart(id){ cart=cart.filter(i=>i.id!==id); updateCartUI(); }
function toggleCart(){ document.getElementById('cart-overlay').classList.toggle('open'); document.getElementById('cart-drawer').classList.toggle('open'); updateCartUI(); }
function submitOrder(){
  if(!verified&&!isOwner){ toggleCart(); openModal(); return; }
  if(!cart.length){ showToast('Your order is empty'); return; }
  const lines=cart.map(i=>`• ${i.qty}x ${i.name} — ${i.price?'$'+(i.price*i.qty).toFixed(2):'Price TBD'}`).join('\n');
  const subject=encodeURIComponent('Order Inquiry – The Queen of Cardboard Collectibles');
  const body=encodeURIComponent(`Hello,\n\nOrder inquiry:\n\n${lines}\n\nThank you!`);
  window.location.href=`mailto:thequeenofcardboard1@gmail.com?subject=${subject}&body=${body}`;
}
